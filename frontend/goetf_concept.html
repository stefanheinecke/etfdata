import requests

API_KEY = "your_key_here"
BASE    = "https://api.goetf.ch/v1"

# Fetch ETF scores
r = requests.get(
    f"{BASE}/scores/etfs",
    headers={"X-API-Key": API_KEY}
)
scores = r.json()
# → [{ticker: "SWDA", score: 7.8, ...}, ...]

# Calculate portfolio overlap
r = requests.post(
    f"{BASE}/analytics/overlap",
    json={"tickers": ["SWDA", "CSSPX", "VUSA"]},
    headers={"X-API-Key": API_KEY}
)
overlap = r.json()
# → {"SWDA_CSSPX": 0.78, "SWDA_VUSA": 0.71, ...}
const API_KEY = "your_key_here";
const BASE    = "https://api.goetf.ch/v1";

// Fetch ETF scores
const scores = await fetch(`${BASE}/scores/etfs`, {
  headers: { "X-API-Key": API_KEY }
}).then(r => r.json());

// Calculate portfolio overlap
const overlap = await fetch(`${BASE}/analytics/overlap`, {
  method: "POST",
  headers: {
    "X-API-Key": API_KEY,
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    tickers: ["SWDA", "CSSPX", "VUSA"]
  })
}).then(r => r.json());
# Fetch ETF scores
curl -H "X-API-Key: your_key_here" \
     https://api.goetf.ch/v1/scores/etfs

# Calculate portfolio overlap
curl -X POST \
     -H "X-API-Key: your_key_here" \
     -H "Content-Type: application/json" \
     -d '{"tickers":["SWDA","CSSPX","VUSA"]}' \
     https://api.goetf.ch/v1/analytics/overlap
